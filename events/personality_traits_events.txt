# @name Lustful/Chaste - male

character_event = {
    id = 1000
    desc = "EVTDESC1000" #A pretty wench is throwing herself at you.
    picture = "GFX_evt_lovers"
    
    min_age = 16
    only_men = yes
    
    is_triggered_only = yes
    
    trigger = {
        NOT = {
            personality_traits = 5
        }
        NOT = {
            # sexless
            OR = {
                culture_group = orc_group
                culture_group = goblin_group
            }
        }
        NOT = {
            trait = chaste
        }
        NOT = {
            trait = lustful
        }
        NOT = {
            trait = celibate
        }
        NOT = {
            trait = homosexual
        }
    }
    
    weight_multiplier = {
        days = 1
        
        modifier = {
            factor = 1.7
            trait = greedy
        }
        modifier = {
            factor = 1.7
            trait = hedonist
        }
        modifier = {
            factor = 1.7
            trait = gluttonous
        }
        modifier = {
            factor = 0.5
            trait = temperate
        }
		#chaos worshippers have greater chances
		modifier = {
			factor = 1.3
			religion_group = chaos_gods_group
		}
		#Slaanesh have even greater chances
		modifier = {
			factor = 1.7
			religion = slaanesh
		}
    }

    option = {
        name = "EVTOPTA1000" #I can't resist her
        if = {
            limit = {
                trait = honest
                NOT = {
                    trait = slothful

            }

                chance = 70
                
                add_trait = lustful

                        id = 38272 #Notify Lustful
                    }
                }
