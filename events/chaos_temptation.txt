    trigger = { 
        is_capable = yes
        OR = { 
            has_character_flag = temptation_tzeenth

            has_character_flag = temptation_khorne
            has_character_flag = temptation_nurgle
        }
        NOT = {
            OR = { 
                has_character_flag = hidden_cultist_nurgle
                has_character_flag = hidden_cultist_slaanesh
                has_character_flag = hidden_cultist_tzeenth
                has_character_flag = hidden_cultist_khorne
            }
        }
    }
    mean_time_to_happen = {
        months = 12
    }
    option = {
        name = "Join the cult"
        if = {
            limit = { has_character_flag = temptation_tzeenth }
            set_character_flag = hidden_cultist_tzeenth
        }
        if = {
            limit = { has_character_flag = temptation_nurgle }
            set_character_flag = hidden_cultist_nurgle 
        }
        if = {
            limit = { has_character_flag = temptation_slaanesh }
            set_character_flag = hidden_cultist_slaanesh
        }
        if = {
            limit = { has_character_flag = temptation_khorne }
            set_character_flag = hidden_cultist_khorne
        }
        clr_character_flag = temptation_tzeenth
        clr_character_flag = temptation_slaanesh
        clr_character_flag = temptation_nurgle
        clr_character_flag = temptation_khorne
        # change doom counter - if big figure became cultist
        if = { 
            limit = { higher_tier_than = COUNT }
            # doom counter +1
            1 = { change_variable = { which = "doom_counter" value = 1 }}
        }

            limit = { higher_tier_than = DUKE }
            # doom counter +1

        }
        if = { 
            limit = { higher_tier_than = KING }
            # doom counter +1

        }

    }
